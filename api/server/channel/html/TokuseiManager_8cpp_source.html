1371 bool TokuseiManager::EvaluateTokuseiCondition(const std::shared_ptr<ActiveEntityState>& eState,
1372  int32_t tokuseiID, const std::shared_ptr<objects::TokuseiCondition>& condition)

1375  condition->GetComparator() != objects::TokuseiCondition::Comparator_t::NOT_EQUAL;
1376

1378  switch(condition->GetType())
1379  {
1380  case TokuseiConditionType::CURRENT_HP:
1381  case TokuseiConditionType::CURRENT_MP:
1382  // Current HP or MP percent matches the comparison type and value
1383  {
1384  auto cs = eState->GetCoreStats();
1385  if(!cs)
1386  {
1387  return false;
1388  }
1389

1391  if(condition->GetType() == TokuseiConditionType::CURRENT_HP)

1394  }
1395  else
1396  {
1397  currentValue = (int32_t)floor((float)cs->GetMP() / (float)eState->GetMaxMP() * 100.f);
1398  }
1399 
1400  return Compare(currentValue, condition, true);
1401  }
1402  break;
1403  case TokuseiConditionType::DIGITALIZED:
1404  // Entity is a character and is digitalized
1405  if(numericCompare ||
1406  eState->GetEntityType() != EntityType_t::CHARACTER)
1407  {
1408  return false;
1409  }
1410  else
1411  {
1412  auto cState = std::dynamic_pointer_cast<CharacterState>(eState);
1413

1415  return digitalized == (condition->GetComparator() ==
1416  objects::TokuseiCondition::Comparator_t::EQUALS);
1417  }
1418  break;
1419  case TokuseiConditionType::EQUIPPED_WEAPON_TYPE:
1420  // Entity is a character and has the specified weapon type equipped
1421  if(numericCompare ||
1422  eState->GetEntityType() != EntityType_t::CHARACTER)
1423  {
1424  return false;
1425  }
1426  else
1427  {
1428  auto cState = std::dynamic_pointer_cast<CharacterState>(eState);
