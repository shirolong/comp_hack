922 uint8_t CharacterState::RecalculateStats(
923  libcomp::DefinitionManager* definitionManager,
924  std::shared_ptr<objects::CalculatedEntityState> calcState,
925  std::shared_ptr<objects::MiSkillData> contextSkill)
926 {
927  uint8_t result = 0;
928 
929  std::lock_guard<std::mutex> lock(mLock);
930 
931  auto c = GetEntity();
932  auto cs = GetCoreStats();
933 
934  bool selfState = calcState == nullptr;
935  if(selfState)
936  {
937  calcState = GetCalculatedState();
938 
939  // Calculate current skills, only matters if calculating
940  // for the default entity state
941  auto previousSkills = GetCurrentSkills();
942  SetCurrentSkills(GetAllSkills(definitionManager, true));
943 
944  bool skillsChanged = previousSkills.size() != CurrentSkillsCount();
945  if(!skillsChanged)
946  {
947  for(uint32_t skillID : previousSkills)
948  {
949  if(!CurrentSkillsContains(skillID))
950  {
951  skillsChanged = true;
952  break;
953  }
954  }
955  }
956  result = skillsChanged ? ENTITY_CALC_SKILL : 0x00;
957 
958  // Remove any switch skills no longer available
959  RemoveInactiveSwitchSkills();
960  }
961 
962  auto stats = CharacterManager::GetCharacterBaseStats(cs);
963 
964  // Adjust base stats based on digitalize
965  auto dgState = mDigitalizeState;
966  if(dgState)
967  {
968  // Apply digitalize non-HP/MP stats (which apply further down)
969  for(auto& pair : dgState->GetCorrectValues())
970  {
971  if(pair.first != (uint8_t)CorrectTbl::HP_MAX &&
972  pair.first != (uint8_t)CorrectTbl::MP_MAX)
973  {
974  stats[(CorrectTbl)pair.first] = (int16_t)(
975  stats[(CorrectTbl)pair.first] + pair.second);
976  }
977  }
978  }
979 
980  if(selfState)
981  {
982  // Combat run speed can change from unadjusted stats (nothing
983  // natively does this)
984  SetCombatRunSpeed((int16_t)stats[CorrectTbl::MOVE2]);
985 
986  if(!mInitialCalc)
987  {
988  SetKnockbackResist((float)stats[CorrectTbl::KNOCKBACK_RESIST]);
989  mInitialCalc = true;
990  }
991  }
992 
993  // Keep track of the current system time for expired equipment
994  uint32_t now = (uint32_t)std::time(0);
995 
996  // Calculate based on adjustments
997  std::list<std::shared_ptr<objects::MiCorrectTbl>> correctTbls;
998  std::list<std::shared_ptr<objects::MiCorrectTbl>> nraTbls;
999  for(size_t i = 0; i < 15; i++)
1000  {
1001  bool bullets = i ==
1002  (size_t)objects::MiItemBasicData::EquipType_t::EQUIP_TYPE_BULLETS;
1003 
1004  auto equip = c->GetEquippedItems(i).Get();
1005  if(equip && (equip->GetDurability() > 0 || bullets) &&
1006  (!equip->GetRentalExpiration() || now < equip->GetRentalExpiration()))
1007  {
1008  uint32_t basicEffect = equip->GetBasicEffect();
1009  auto itemData = definitionManager->GetItemData(
1010  basicEffect ? basicEffect : equip->GetType());
1011  for(auto ct : itemData->GetCommon()->GetCorrectTbl())
1012  {
1013  if((uint8_t)ct->GetID() >= (uint8_t)CorrectTbl::NRA_WEAPON &&
1014  (uint8_t)ct->GetID() <= (uint8_t)CorrectTbl::NRA_MAGIC)
1015  {
1016  nraTbls.push_back(ct);
1017  }
1018  else
1019  {
1020  correctTbls.push_back(ct);
1021  }
1022  }
1023  }
1024  }
1025 
1026  if(dgState)
1027  {
1028  // Digitalize passives are "floating" and not directly on the character
1029  auto dgPassives = dgState->GetPassiveSkills();
1030  ApplySkillCorrectTbls(dgPassives, definitionManager, correctTbls);
1031  }
1032 
1033  GetAdditionalCorrectTbls(definitionManager, calcState, correctTbls,
1034  contextSkill);
1035 
1036  UpdateNRAChances(stats, calcState, nraTbls);
1037  AdjustStats(correctTbls, stats, calcState, true);
1038 
1039  // Base stats calcualted, Apply equipment fusion bonuses now
1040  for(auto& pair : mEquipFuseBonuses)
1041  {
1042  stats[pair.first] = (int16_t)(stats[pair.first] + pair.second);
1043  }
1044 
1045  CharacterManager::CalculateDependentStats(stats, cs->GetLevel(), false);
1046 
1047  if(dgState)
1048  {
1049  // Add digitalize HP/MP now
1050  stats[CorrectTbl::HP_MAX] = (int16_t)(stats[CorrectTbl::HP_MAX] +
1051  dgState->GetCorrectValues((uint8_t)CorrectTbl::HP_MAX));
1052  stats[CorrectTbl::MP_MAX] = (int16_t)(stats[CorrectTbl::MP_MAX] +
1053  dgState->GetCorrectValues((uint8_t)CorrectTbl::MP_MAX));
1054  }
1055 
1056  if(selfState)
1057  {
1058  result = result | CompareAndResetStats(stats, true);
1059  }
1060 
1061  AdjustStats(correctTbls, stats, calcState, false);
1062 
1063  if(GetStatusTimes(STATUS_RESTING))
1064  {
1065  // Apply (originally busted) Medical Sciences bonus of 10% more
1066  // regen per class
1067  uint8_t cls = (uint8_t)(GetExpertiseRank(EXPERTISE_MEDICAL_SCIENCES) /
1068  10);
1069  if(cls)
1070  {
1071  stats[CorrectTbl::HP_REGEN] = (int16_t)((double)
1072  stats[CorrectTbl::HP_REGEN] * (1.0 + 0.1 * (double)cls));
1073  stats[CorrectTbl::MP_REGEN] = (int16_t)((double)
1074  stats[CorrectTbl::MP_REGEN] * (1.0 + 0.1 * (double)cls));
1075  }
1076  }
1077 
1078  if(selfState)
1079  {
1080  return result | CompareAndResetStats(stats, false);
1081  }
1082  else
1083  {
1084  for(auto statPair : stats)
1085  {
1086  calcState->SetCorrectTbl((size_t)statPair.first,
1087  (int16_t)statPair.second);
1088  }
1089 
1090  return result;
1091  }
1092 }
