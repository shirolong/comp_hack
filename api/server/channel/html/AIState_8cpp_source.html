103 bool AIState::SetStatus(AIStatus_t status, bool isDefault)
104 {
105  // Disallow default setting to target dependent status
106  if(isDefault &&
107  (status == AIStatus_t::AGGRO || status == AIStatus_t::COMBAT ||
108  status == AIStatus_t::FOLLOWING))
109  {
110  return false;
111  }
112 
113  bool statusChanged = false;
114  {
115  std::lock_guard<std::mutex> lock(mFieldLock);
116  mStatusChanged = statusChanged = mStatus != status;
117 
118  mPreviousStatus = mStatus;
119  mStatus = status;
120 
121  if(isDefault)
122  {
123  mDefaultStatus = status;
124  }
125  }
126 
127  if(statusChanged)
128  {
129  // Always reset next target time
130  SetNextTargetTime(0);
131 
132  if(status == AIStatus_t::WANDERING)
133  {
134  uint64_t now = ChannelServer::GetServerTime();
135  if(GetDespawnWhenLost() && !HasFollowTarget())
136  {
137  // Most entities despawn when switching to wandering after a
138  // set time if they don't make their way back to their spawn
139  // location
140  SetDespawnTimeout(now + (uint64_t)AI_DESPAWN_TIMEOUT);
141  }
142 
143  // Set next target time based on think speed
144  SetNextTargetTime(now + (uint64_t)(GetThinkSpeed() * 1000));
145  }
146  else if(GetDespawnTimeout() && !HasFollowTarget())
147  {
148  // Clear if switching to anything else
149  SetDespawnTimeout(0);
150  }
151  }
152 
153  return true;
154 }
