429 std::list<std::shared_ptr<objects::ClientCostAdjustment>>
430  ClientState::SetCostAdjustments(int32_t entityID, std::list<
431  std::shared_ptr<objects::ClientCostAdjustment>> adjustments)
432 {
433  std::list<std::shared_ptr<objects::ClientCostAdjustment>> updates;
434 
435  std::lock_guard<std::mutex> lock(mLock);
436 
437  // Store old and set new
438  auto current = mCostAdjustments[entityID];
439  if(adjustments.size() == 0)
440  {
441  // No new adjustments
442  mCostAdjustments.erase(entityID);
443 
444  if(current.size() == 0)
445  {
446  // No updates
447  return updates;
448  }
449  }
450  else
451  {
452  // New adjustments exist
453  mCostAdjustments[entityID] = adjustments;
454  }
455 
456  // Compare and set updates
457  for(auto adjust : adjustments)
458  {
459  std::shared_ptr<objects::ClientCostAdjustment> match;
460  for(auto it = current.begin(); it != current.end(); it++)
461  {
462  if((*it)->GetCategory() == adjust->GetCategory() &&
463  (*it)->GetType() == adjust->GetType())
464  {
465  match = *it;
466  current.erase(it);
467  break;
468  }
469  }
470 
471  if(match)
472  {
473  if(match->GetHPCost() != adjust->GetHPCost() ||
474  match->GetMPCost() != adjust->GetMPCost())
475  {
476  // Updated
477  updates.push_back(adjust);
478  }
479  }
480  else if(adjust->GetHPCost() != 100 || adjust->GetMPCost() != 100)
481  {
482  // Newly set
483  updates.push_back(adjust);
484  }
485  }
486 
487  // Create removals from leftover changes
488  for(auto c : current)
489  {
490  if(c->GetHPCost() != 100 || c->GetMPCost() != 100)
491  {
492  auto defaultCost = std::make_shared<objects::ClientCostAdjustment>();
493  defaultCost->SetCategory(c->GetCategory());
494  defaultCost->SetType(c->GetType());
495  updates.push_back(defaultCost);
496  }
497  }
498 
499  return updates;
500 }
