445 void AIManager::CombatSkillComplete(
446  const std::shared_ptr<ActiveEntityState>& eState,
447  const std::shared_ptr<objects::ActivatedAbility>& activated,
448  const std::shared_ptr<objects::MiSkillData>& skillData,
449  const std::shared_ptr<ActiveEntityState>& target, bool hit)
450 {
451  auto aiState = eState->GetAIState();
452  if(!aiState)
453  {
454  return;
455  }
456 
457  // Multiple triggers in combat cause normal AI to reset and reorient
458  // itself so they're not spamming skills non-stop
459  bool reset = false;
460  bool wait = true;
461 
462  bool normalProcesing = true;
463  if(aiState->ActionOverridesKeyExists("combatSkillComplete"))
464  {
465  libcomp::String fOverride = aiState->GetActionOverrides(
466  "combatSkillComplete");
467 
468  LogAIManagerDebug([eState, fOverride]()
469  {
470  return libcomp::String("Executing combatSkillComplete override for"
471  " %1: %2\n").Arg(eState->GetEntityLabel())
472  .Arg(fOverride);
473  });
474 
475  Sqrat::Function f(Sqrat::RootTable(aiState->GetScript()->GetVM()),
476  fOverride.IsEmpty() ? "combatSkillComplete" : fOverride.C());
477 
478  auto scriptResult = !f.IsNull()
479  ? f.Evaluate<int32_t>(eState, this, activated, target, hit) : 0;
480  if(!scriptResult || *scriptResult == -1)
481  {
482  // Do not continue
483  return;
484  }
485  else if(*scriptResult & 0x01)
486  {
487  // Skip normal processing
488  normalProcesing = false;
489  reset = (*scriptResult & 0x02) != 0;
490  wait = (*scriptResult & 0x04) == 0;
491  }
492  }
493 
494  if(normalProcesing && target)
495  {
496  if(target->GetStatusTimes(STATUS_KNOCKBACK))
497  {
498  // If the target is currently being knocked back (from this
499  // skill or some other one), reset
500  reset = true;
501  }
502  else if(eState->GetStatusTimes(STATUS_HIT_STUN))
503  {
504  // If the source is hitstunned for whatever reason (counter
505  // or guard for example)
506  reset = true;
507  }
508  else if(!skillData->GetTarget()->GetRange() &&
509  !skillData->GetCast()->GetBasic()->GetChargeTime() &&
510  !skillData->GetCondition()->GetCooldownTime())
511  {
512  // No charge, no cooldown, no range combat skills are typically
513  // used in succession until knockback occurs (delayed by lockout
514  // animation time)
515  bool combo = false;
516 
517  if(target->GetStatusTimes(STATUS_HIT_STUN))
518  {
519  // If the target is hitstunned, use again to attempt to combo
520  // into knockback most of the time. Even if no combo occurs,
521  // do not wait to use the next skill.
522  combo = RNG(int32_t, 1, 10) <= 9;
523  wait = false;
524  }
525  else
526  {
527  // If the target was still hit, repeat attack 30% of the
528  // time, 10% if they were not hit
529  combo = (hit && RNG(int32_t, 1, 10) <= 3) ||
530  (!hit && RNG(int32_t, 1, 10) == 1);
531  }
532 
533  if(combo && !aiState->GetCurrentCommand())
534  {
535  auto cmd = std::make_shared<AIUseSkillCommand>(skillData,
536  target->GetEntityID());
537  aiState->QueueCommand(cmd);
538  }
539  else
540  {
541  reset = true;
542  }
543  }
544  else
545  {
546  // Check what kind of skill it was to decide how to handle
547  switch(skillData->GetBasic()->GetActionType())
548  {
549  case objects::MiSkillBasicData::ActionType_t::GUARD:
550  case objects::MiSkillBasicData::ActionType_t::COUNTER:
551  case objects::MiSkillBasicData::ActionType_t::DODGE:
552  // Reset actions but do not wait
553  reset = true;
554  wait = false;
555  break;
556  default:
557  // Other skills should be staggered unless more
558  // executions exist
559  reset = activated->GetExecuteCount() >= activated
560  ->GetMaxUseCount();
561  break;
562  }
563  }
564  }
565 
566  if(reset)
567  {
568  aiState->ClearCommands();
569  if(wait)
570  {
571  auto logicGroup = aiState->GetLogicGroup();
572  if(logicGroup && logicGroup->GetSkillResetStagger())
573  {
574  QueueWaitCommand(aiState, logicGroup->GetSkillResetStagger());
575  }
576  }
577  }

925 bool AIManager::Circle(const std::shared_ptr<ActiveEntityState>& eState,
926  float x, float y, bool interrupt, float distance)
927 {
928  auto zone = eState ? eState->GetZone() : nullptr;
929  auto aiState = eState ? eState->GetAIState() : nullptr;
930  if(!zone || !aiState || !eState->CanMove())
931  {
932  return false;
933  }
934 
935  auto server = mServer.lock();
936  auto zoneManager = server->GetZoneManager();
937 
938  Point src(eState->GetCurrentX(), eState->GetCurrentY());
939  Point target(x, y);
940 
941  std::list<Point> pathing;
942 
943  // Distance from the target first
944  Point start = zoneManager->GetLinearPoint(target.x, target.y,
945  src.x, src.y, distance, false, zone);
946  if(start != src)
947  {
948  // If we can't get to the start point in a straight line
949  // quit now as the move should not be that complex
950  Point collidePoint;
951  if(zone->Collides(Line(start, src), collidePoint))
952  {
953  return false;
954  }
955 
956  pathing.push_back(start);
957  }
958 
959  // Rotate the starting point around the target for the second
960  // (or third) point
961  int32_t pointCount = RNG(int32_t, 1, 2);
962  bool invert = RNG(int32_t, 1, 2) == 1;
963  for(int32_t i = 0; i < pointCount; i++)
964  {
965  Point prev = pathing.size() > 0 ? pathing.back() : src;
966  Point p = zoneManager->RotatePoint(prev, target,
967  invert ? -0.52f : 0.52f);
968 
969  // Check for collision
970  Point pFinal = zoneManager->GetLinearPoint(prev.x, prev.y,
971  p.x, p.y, prev.GetDistance(p), false, zone);
972  if(pFinal != p && i == 0)
973  {
974  // Hit something, try the other direction (only once)
975  pointCount++;
976  invert = !invert;
977  }
978  else if(pFinal == prev)
979  {
980  // Can't move further
981  break;
982  }
983  else
984  {
985  // Add the point
986  pathing.push_back(pFinal);
987  }
988  }
989 
990  if(pathing.size() > 0)
991  {
992  auto cmd = std::make_shared<AIMoveCommand>();
993  cmd->SetPathing(pathing);
994  aiState->QueueCommand(cmd, interrupt);
995 
996  return true;
997  }
998 
999  return false;
