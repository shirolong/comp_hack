818 const WorldClock ChannelServer::GetWorldClockTime()
819 {
820  // World time is relative to seconds so no need for precision past time_t.
821  time_t systemTime = std::time(0);
822 
823  // If the system time has not been updated, no need to run the
824  // calculation again
825  if((uint32_t)systemTime == mWorldClock.SystemTime)
826  {
827  return mWorldClock;
828  }
829 
830  std::lock_guard<std::mutex> lock(mTimeLock);
831 
832  // Check again after lock so we don't double calculate
833  if((uint32_t)systemTime == mWorldClock.SystemTime)
834  {
835  return mWorldClock;
836  }
837 
838  bool eventPassed = mWorldClock.SystemTime < mNextEventTime &&
839  mNextEventTime <= (uint32_t)systemTime;
840 
841  // Set the GMT system time
842  WorldClock newClock;
843  newClock.SystemTime = (uint32_t)systemTime;
844 
845  // Adjust the time offset to get relative real time values
846  int32_t offset = GetServerTimeOffset();
847  if(offset)
848  {
849  systemTime = (systemTime + offset);
850  }
851 
852  tm* t = gmtime(&systemTime);
853 
854  newClock.WeekDay = (int8_t)(t->tm_wday + 1);
855  newClock.Month = (int8_t)(t->tm_mon + 1);
856  newClock.Day = (int8_t)t->tm_mday;
857  newClock.SystemHour = (int8_t)t->tm_hour;
858  newClock.SystemMin = (int8_t)t->tm_min;
859  newClock.SystemSec = (int8_t)t->tm_sec;
860 
861  // Now calculate the game relative times
862  newClock.GameOffset = mWorldClock.GameOffset;
863 
864  // Every 4 days, 15 full moon cycles will elapse and the same game time
865  // will occur on the same time offset.
866  newClock.CycleOffset = (uint32_t)((newClock.SystemTime +
867  newClock.GameOffset - BASE_WORLD_TIME) % 345600);
868 
869  // 24 minutes = 1 game phase (16 total)
870  newClock.MoonPhase = (int8_t)((newClock.CycleOffset / 1440) % 16);
871 
872  // 2 minutes = 1 game hour
873  newClock.Hour = (int8_t)((newClock.CycleOffset / 120) % 24);
874 
875  // 2 seconds = 1 game minute
876  newClock.Min = (int8_t)((newClock.CycleOffset / 2) % 60);
877 
878  // Replace the old clock values
879  mWorldClock = newClock;
880 
881  if(eventPassed || mNextEventTime == 0)
882  {
883  mRecalcTimeDependents = true;
884  RecalcNextWorldEventTime();
885  }
886 
887  return newClock;
888 }
