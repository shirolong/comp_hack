112  default:

354 void ManagerConnection::HandleClientTimeouts(uint64_t now, uint16_t timeout)
355 {
356  std::list<libcomp::String> timeOuts;
357  {
358  ServerTime expireBefore = now - (ServerTime)(timeout * 1000000ULL);

387  p.WriteU32Little((uint32_t)LogoutPacketAction_t::LOGOUT_DISCONNECT);
