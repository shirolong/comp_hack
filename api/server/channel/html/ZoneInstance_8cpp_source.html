331  ZoneInstance::UpdateDestinyBox(int32_t worldCID, uint8_t& newNext,
332  const std::list<std::shared_ptr<objects::Loot>> &add,
333  const std::set<uint8_t>& remove)
334 {

336 
337  newNext = 0;
338 
339  auto dBox = GetDestinyBox(worldCID);
340  if(!dBox)
341  {
342  return results;
343  }
344 
345  std::lock_guard<std::mutex> lock(mLock);
346 
347  size_t size = dBox->LootCount();
348 
349  // Do removes first (append to end to "shift" forward)
350  for(uint8_t slot : remove)
351  {
352  if(dBox->GetLoot(slot))
353  {
354  dBox->InsertLoot(slot, nullptr);
355  dBox->RemoveLoot((size_t)(slot + 1));
356  results[slot] = nullptr;
357  }
358  }
359 
360  // Update the next position
361  newNext = dBox->GetNextPosition();
362  if(remove.size() > 0)
363  {
364  // Always maximize the amount of spaces between next and the first
365  // item that will be overwritten based on the current position
366  size_t seen = 0;
367  while(!dBox->GetLoot(newNext) && seen < size)
368  {
369  uint8_t previous = 0;
370  if(newNext == 0)
371  {
372  previous = (uint8_t)(size - 1);
373  }
374  else
375  {
376  previous = (uint8_t)((size_t)(newNext - 1) % size);
377  }
378 
379  if(!dBox->GetLoot(previous))
380  {
381  newNext = previous;
382  seen++;
383  }
384  else
385  {
386  break;
387  }
388  }
389 
390  if(dBox->GetLoot(newNext))
391  {
392  // Removed elements did not include starting next, jump forward
393  // to the first empty element
394  for(size_t i = 0; i < size; i++)
395  {
396  size_t slot = (size_t)((newNext + i) % size);
397  if(!dBox->GetLoot(slot))
398  {
399  newNext = (uint8_t)slot;
400  break;
401  }
402  }
403  }
404  else if(seen == size)
405  {
406  // All removed, reset to start
407  newNext = 0;
408  }
409  }
410 
411  // Next do updates
412  for(auto loot : add)
413  {
414  dBox->InsertLoot(newNext, loot);
415  dBox->RemoveLoot((size_t)(newNext + 1));
416  results[newNext] = loot;
417 
418  newNext = (uint8_t)(newNext + 1);
419  if(newNext >= (uint8_t)size)
420  {
421  newNext = 0;
422  }
423  }
424 
425  // Set the new next position

427
