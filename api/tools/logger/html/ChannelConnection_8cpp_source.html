614 bool ChannelConnection::packetHasServerSwitch(libcomp::Packet& p)
615 {
616  // This is how big a "gzip" packet has to be
617  // to read everything before the actual data.
618  if(p.Size() < 24)
619  return false;
620 
621  // Make sure we are at the right spot.
622  p.Rewind();
623 
624  // Read the sizes of the packet.
625  uint32_t paddedSize = p.ReadU32Big();
626  uint32_t realSize = p.ReadU32Big();
627 
628  // Calculate how much data is padding.
629  uint32_t padding = paddedSize - realSize;
630 
631  // Read the gzip "magic".
632  uint32_t gzip = p.ReadU32Big();
633 
634  // All packets start with this.
635  if(gzip != 0x677A6970)
636  return false;
637 
638  // Get the uncompressed and compressed size.
639  int32_t uncompressed_size = p.ReadS32Little();
640  int32_t compressed_size = p.ReadS32Little();
641 
642  // Get the compression level.
643  uint32_t lv6 = p.ReadU32Big();
644 
645  // The compression level should always be the same.
646  if(lv6 != 0x6C763600)
647  return false;
648 
649  // Make sure we have enough data left to decompress the entire packet.
650  if(p.Left() != (static_cast<uint32_t>(compressed_size) + padding))
651  return false;
652 
653  // If the compressed and uncompressed sizes differ, decompress the packet.
654  if(compressed_size != uncompressed_size)
655  {
656  // Decompress the packet and get the uncompressed size.
657  int32_t sz = p.Decompress(compressed_size);
658 
659  // Check the uncompressed size matches the recorded size.
660  if(sz != uncompressed_size)
661  return false;
662 
663  // There is no padding anymore.
664  padding = 0;
665  }
666 
667  // Default to no switch packet.
668  bool hasSwitchPacket = false;
669 
670  // Loop through and check each command.
671  while(p.Left() > padding)
672  {
673  // Make sure there is enough data.
674  if(p.Left() < 6)
675  return false;
676 
677  // Skip over the big endian size.
678  p.Skip(2);
679 
680  // Read the command start, size, and code.
681  uint32_t cmd_start = p.Tell();
682  uint16_t cmd_size = p.ReadU16Little();
683  uint16_t code = p.ReadU16Little();
684 
685  // With no data, the command size is 4 bytes.
686  if(cmd_size < 4)
687  return false;
688 
689  // Check there is enough data left.
690  if(p.Left() < (uint32_t)(cmd_size - 4))
691  return false;
692 
693  // Read the command.
694  libcomp::Packet cmd(p.Data() + cmd_start + 4, cmd_size - 4u);
695 
696  // Check for the switch command.
697  if(code == 0x0009 && cmd.ReadU32Little() == 14)
698  hasSwitchPacket = true;
699 
700  // Move to the next command.
701  p.Seek(cmd_start + cmd_size);
702  } // while(p.Left() > padding)
703 
704  // Skip the padding.
705  p.Skip(padding);
706 
707  // Check that the entire packet was read.
708  if(p.Left() != 0)
709  return false;
710 
711  return hasSwitchPacket;
712 }

754  // Decompress the packet and get the uncompressed size.
755  int32_t sz = p.Decompress(compressed_size);
756 
757  // Check the uncompressed size matches the recorded size.
758  if(sz != uncompressed_size)
759  return;
760 
761  // There is no padding anymore.
762  padding = 0;
763  }
764 
765  // Create a buffer to write the commands into.
766  libcomp::Packet packetQueue;
767 
768  // Add the generic header information to the buffer.
769  packetQueue.WriteBlank(8);
770  packetQueue.WriteArray("gzip", 4);
771  packetQueue.WriteBlank(8);
772  packetQueue.WriteArray("lv6", 4);
773 
774  // Loop through and check each command.
775  while(p.Left() > padding)
776  {
777  // Make sure there is enough data.
778  if(p.Left() < 6)
779  return;
780 
781  // Skip over the big endian size.
782  p.Skip(2);
783 
784  // Read the command start, size, and code.
785  uint32_t cmd_start = p.Tell();
786  uint16_t cmd_size = p.ReadU16Little();
787  uint16_t code = p.ReadU16Little();
788 
789  // With no data, the command size is 4 bytes.
790  if(cmd_size < 4)
791  return;
792 
793  // Check there is enough data left.
794  if(p.Left() < (uint32_t)(cmd_size - 4))
795  return;
796 
797  // Read the command.
798  libcomp::Packet cmd(p.Data() + cmd_start + 4, cmd_size - 4u);
799 
800  // Check for the switch command.
801  if(code == 0x0009 && cmd.ReadU32Little() == 14)
802  {
803  // Seed over the data we don't care about.
804  cmd.Seek(4);
805 
806  // Read in the session key.
807  uint32_t sessionKey = cmd.ReadU32Little();
808 
809  // Read the original address.
810  libcomp::String origAddr = cmd.ReadString16Little(
811  libcomp::Convert::ENCODING_UTF8);
812 
813  // Save the original info.
814  mServer->registerChannelKey(sessionKey, origAddr.C());
815 
816  // Generate the new address.
817  QString addr = QString("%1:14666").arg(
818  mClientSocket->localAddress().toString() );
819 
820  // Log the message.
821  logMessage(tr("Sending client to relay..."));
822  logMessage(tr("Gave the client: %2").arg(addr));
823  logMessage(tr("Original address: %2").arg(origAddr.C()));
824 
825  // Adjust the command size.
826  uint16_t new_size = (uint16_t)(cmd_size - origAddr.Length() +
827  static_cast<uint32_t>(addr.length()));
828 
829  // Write the new server switch command.
830  packetQueue.WriteU16Big(new_size);
831  packetQueue.WriteU16Little(new_size);
832  packetQueue.WriteU16Little(code);
833  packetQueue.WriteU32Little(14);
834  packetQueue.WriteU32Little(sessionKey);
835  packetQueue.WriteString16Little(libcomp::Convert::ENCODING_UTF8,
836  addr.toUtf8().constData(), true);
837  }
838  else
839  {
840  // Write the command without modification.
841  packetQueue.WriteU16Big(cmd_size);
842  packetQueue.WriteU16Little(cmd_size);
843  packetQueue.WriteU16Little(code);
844  packetQueue.WriteArray(cmd.Data(), cmd.Size());
845  }
846 
847  // Move to the next command.
848  p.Seek(cmd_start + cmd_size);
849  } // while(p.Left() > padding)
850 
851  // Skip the padding.
852  p.Skip(padding);
853 
854  // Check that the entire packet was read.
855  if(p.Left() != 0)
856  return;
857 
858  // Calculate the new packet size.
859  uint32_t packetSize = packetQueue.Size() - 24;
860 
861  // Calculate the real and padded size.
862  realSize = packetSize + 16;
863  paddedSize = realSize - (realSize % 8);
864 
865  // Adjust the padded size.
866  if(realSize != paddedSize)
867  paddedSize += 8;
868 
869  // Rewind and fix the sizes for the packet.
870  packetQueue.Rewind();
871  packetQueue.WriteU32Big(paddedSize);
872  packetQueue.WriteU32Big(realSize);
873  packetQueue.Skip(4);
874  packetQueue.WriteU32Little(packetSize); // Uncompressed
875  packetQueue.WriteU32Little(packetSize); // Compressed
876 
877  // Compress the packet if the original packet was compressed.
878  if(compressed_size != uncompressed_size)
879  {
880  // Seek to the data in the packet and compress the packet.
881  packetQueue.Seek(24);
882  packetQueue.Compress(static_cast<int32_t>(packetSize));
883 
884  // If they are equal, this packet might be
885  // confused with an uncompressed one.
886  if((packetQueue.Size() - 24) == packetSize)
887  return;
888 
889  // Update the compressed size.
890  packetSize = packetQueue.Size() - 24;
891 
892  // Write the compressed size.
893  packetQueue.Seek(16);
894  packetQueue.WriteU32Little(packetSize);
895 
896  // Update the padded and real size.
897  realSize = packetSize + 16;
898  paddedSize = realSize - (realSize % 8);
899 
900  // Adjust the padded size.
901  if(realSize != paddedSize)
902  paddedSize += 8;
903 
904  // Write the new sizes.
905  packetQueue.Rewind();
906  packetQueue.WriteU32Big(paddedSize);
907  packetQueue.WriteU32Big(realSize);
908  }
909 
910  // Add in the padding.
911  if(realSize != paddedSize)
912  {
913  packetQueue.End();
914  packetQueue.WriteBlank(paddedSize - realSize);
915  }
916 
917  // Rewind to the beginning of the packet.
918  packetQueue.Rewind();
919 
920  // Replace the packet.
921  p.Clear();
922  p.Rewind();
923  p.WriteArray(packetQueue.Data(), packetQueue.Size());
924  p.Rewind();
925 }

927 void ChannelConnection::exchangeKeys()
928 {
929  libcomp::Packet p;
930 
931  // Read in the exchange packet.
932  mServerSocket->read(p.Direct(529), 529);
933 
934  // Skip over the zero value.
935  p.Seek(4);
936 
937  // Read in the server encryption data.
938  mServerCryptData.base = p.ReadString32Big(
939  libcomp::Convert::ENCODING_UTF8);
940  mServerCryptData.prime = p.ReadString32Big(
941  libcomp::Convert::ENCODING_UTF8);
942  mServerCryptData.serverPublic = p.ReadString32Big(
943  libcomp::Convert::ENCODING_UTF8);
944 
945  // If the packet didn't read write, disconnect from the server.
946  if(p.Left() != 0)
947  return serverLost();
948 
949  // Generate the client public to send to the server.
950  mServerCryptData.secret = libcomp::Crypto::GenerateRandom();
951  mServerCryptData.clientPublic = libcomp::Crypto::GenDiffieHellman(
952  mServerCryptData.base, mServerCryptData.prime,
953  mServerCryptData.secret);
954 
955  // Generate the shared secret based on the data from the server.
956  mServerCryptData.sharedKey = libcomp::Crypto::GenDiffieHellman(
957  mServerCryptData.serverPublic, mServerCryptData.prime,
958  mServerCryptData.secret);
959  mServerCryptData.keys = QByteArray::fromHex(
960  mServerCryptData.sharedKey.C());
961 
962  // Set the shared encryption key.
963  BF_set_key(&mServerCryptData.key, 8,
964  (uchar*)mServerCryptData.keys.constData());
965 
966  // Send the client side of the key exchange.
967  {
968  libcomp::Packet reply;
969 
970  // Write the client public to the packet.
971  reply.WriteString32Big(libcomp::Convert::ENCODING_UTF8,
972  mServerCryptData.clientPublic);
973 
974  // Send the packet.
975  mServerSocket->write((char*)reply.Data(), reply.Size());
976  mServerSocket->flush();
977  }
978 
979  // Send login packet.
980  if(mClientLoginPacket)
981  {
982  libcomp::Packet reply;
983 
984  // Write the login packet.
985  reply.WriteArray(mClientLoginPacket, mClientLoginPacketSize);
986 
987  // Encrypt the packet.
988  encryptPacket(&mServerCryptData.key, reply);
989 
990  // Send the packet.
991  mServerSocket->write((char*)reply.Data(), reply.Size());
992  mServerSocket->flush();
993  }
994 
995  // Send packets that came in since the hello.
996  foreach(QByteArray oldPacket, mPacketBuffer)
997  {
998  libcomp::Packet copy;
999 
1000  // Write the packet.
1001  copy.WriteArray(oldPacket.constData(), static_cast<uint32_t>(
1002  oldPacket.size()));
1003 
1004  // Encrypt the packet.
1005  encryptPacket(&mServerCryptData.key, copy);
1006 
1007  // Send the packet.
1008  mServerSocket->write(copy.Data(), copy.Size());
1009  mServerSocket->flush();
1010  }
1011 
1012  // Clear the packet buffer.
1013  mPacketBuffer.clear();
1014 
1015  // Set the server state.
1016  mServerState = Encrypted;
1017 
1018  // Free the login packet.
1019  delete mClientLoginPacket;
1020  mClientLoginPacket = 0;
1021 }
