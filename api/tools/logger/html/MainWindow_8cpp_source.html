256 void MainWindow::patchWebAccess(const QString& path)
257 {
258  QString webAccessPath = tr("%1/webaccess.sdat").arg(path);
259 
260  // Decrypt and load the file into a buffer.
261  std::vector<char> webAccessData = libcomp::Crypto::DecryptFile(
262  webAccessPath.toUtf8().constData());
263  if(webAccessData.empty())
264  return;
265 
266  // Convert the file to a string for processing.
267  QString xml = QString::fromUtf8(&webAccessData[0], static_cast<int>(
268  webAccessData.size()));
269 
270  // Replace the login URL with the logger.
271  xml.replace( QRegExp("\\<login\\s*\\=\\s*[^\\>]+\\>"),
272  tr("<login = http://%1:10999/>").arg(serverLine()) );
273 
274  // Convert the string back into binary.
275  QByteArray out = xml.toUtf8();
276 
277  // Encrypt the data into a buffer.
278  libcomp::Crypto::EncryptFile(webAccessPath.toUtf8().constData(),
279  std::vector<char>(out.data(), out.data() + out.size()));
280 }

282 QList<quint32> MainWindow::findMatches(const void *haystack,
283  quint32 haystack_size, const void *needle, quint32 needle_size)
284 {
285  QList<quint32> matches;
286 
287  haystack_size -= needle_size;
288 
289  const uchar *d = static_cast<const uchar*>(haystack);
290 
291  // Search for a needle in a haystack. Add all occurrences of the needle
292  // to the list "matches".
293  for(quint32 i = 0; i < haystack_size; i++)
294  {
295  if(memcmp(d + i, needle, needle_size) != 0)
296  continue;
297 
298  matches << static_cast<quint32>(i);
299  }
300 
301  return matches;

304 bool MainWindow::versionCheck(const QString& path,
305  float *floatVer, uint32_t *ver, bool *isUS)
306 {
307  // Open the client executable.
308  QFile inFile(path);
309  if( !inFile.open(QIODevice::ReadOnly) )
310  return false;
311 
312  // Map the client executable into virtual memory.
313  uint32_t inSize = (uint32_t)inFile.size();
314  uchar *inMap = inFile.map(0, inSize, QFile::NoOptions);
315 
316  if(!inMap)
317  return false;
318 
319  libcomp::PEFile pe(inMap);
320 
321  QList<quint32> matches;
322 
323  // Find the US version string.
324  matches = findMatches(inMap, inSize, "IMAGINE Version %4.3fU", 22);
325 
326  // If the US version string was found, the client is US; otherwise, search
327  // for the JP version string. If neither is found, the client is invalid.
328  // If the JP version string was found, the client is JP.
329  if(matches.count() != 1)
330  {
331  matches = findMatches(inMap, inSize, "IMAGINE Version %4.3f", 21);
332  if(matches.count() != 1)
333  return false;
334 
335  if(isUS)
336  *isUS = false;
337  }
338  else if(isUS)
339  {
340  *isUS = true;
341  }
342 
343  // Get the offset of the version string in the executable and convert it to
344  // a virtual address.
345  uint32_t verStrOffset = matches.first();
346  uint32_t verStrAddress = pe.offsetToAddress(verStrOffset, ".rwdata");
347 
348  // Search for the address to find the code that references the string.
349  matches = findMatches(inMap, inSize, &verStrAddress, 4);
350  if(matches.count() != 1)
351  return false;
352 
353  // Check for all the expected commands to make sure the section of code is
354  // still valid.
355  uint32_t pushOffset = matches.first() - 1;
356 
357  // Check for push.
358  if(*(inMap + pushOffset) != 0x68)
359  return false;
360 
361  bool isDouble = false;
362 
363  // Check for fld.
364  if(*(inMap + pushOffset - 16) != 0xD9 ||
365  *(inMap + pushOffset - 15) != 0x05)
366  {
367  // Check for a double instead.
368  if(*(inMap + pushOffset - 16) != 0xDD)
369  return false;
370  if(*(inMap + pushOffset - 15) != 0x05)
371  return false;
372 
373  isDouble = true;
374  }
375 
376  // The version is either stored as a double or float. Either way, take the
377  // address of the constant that needs to be loaded and convert it to a file
378  // offset. Next, read that offset and store the resulting version number.
379  uint32_t verAddress;
380  memcpy(&verAddress, inMap + pushOffset - 14, 4);
381 
382  float version = 0.0f;
383  if(isDouble)
384  {
385  uint32_t verOffset = pe.absoluteToOffset(verAddress, ".rdata");
386  if(verOffset > inSize)
387  return false;
388 
389  double doubleVer;
390  memcpy(&doubleVer, inMap + verOffset, 8);
391 
392  version = (float)doubleVer;
393  }
394  else
395  {
396  uint32_t verOffset = pe.addressToOffset(verAddress, ".rwdata");
397  if(verOffset > inSize)
398  return false;
399 
400  memcpy(&version, inMap + verOffset, 4);
401  }
402 
403  // Fill in the requested variables.
404  if(ver)
405  *ver = (uint32_t)(version * 1000.0f + 0.5f);
406  if(floatVer)
407  *floatVer = version;
408 
409  // Unmap and close the file.
410  inFile.unmap(inMap);
411  inFile.close();
412 
413  return true;
414 }
