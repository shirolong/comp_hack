367  avail -= 260;
368

370  // return from the function and wait for more data.
371  if(avail < 8)

373  }

376  libcomp::Packet p;

379  mClientSocket->peek(p.Direct(8), 8);

756 void LobbyConnection::logPacket(libcomp::Packet& p, uint8_t source)
757 {
758  // 0 - Packet came from client.
759  // 1 - Packet came from server.
760 
761  // Only bother if the log file is open.
762  if(!mCaptureLog.isOpen())
763  return;
764 
765  // Rewind to the beginning of the packet.
766  p.Rewind();
767 
768  // Read in the size of the packet.
769  uint32_t size = p.ReadU32Big() + 8;
770 
771  // If there is no size, assume the size is the entire packet.
772  if(size == 0)
773  size = p.Size();
774 
775  // Get the current time.
776  time_t stamp = QDateTime::currentDateTime().toTime_t();
777 
778  // Write the packet to the log.
779  mCaptureLog.write((char*)&source, sizeof(source));
780  mCaptureLog.write((char*)&stamp, 4);
781  mCaptureLog.write((char*)&size, sizeof(size));
782  mCaptureLog.write(p.Data(), size);
783 
784  // Rewind to the beginning of the packet.
785  p.Rewind();
786 }

788 void LobbyConnection::parseStartGamePacket(libcomp::Packet& p)
789 {
790  // Read the session key.
791  uint32_t sessionKey = p.ReadU32Little();
792 
793  // Read the original address.
794  libcomp::String origAddr = p.ReadString16Little(
795  libcomp::Convert::ENCODING_UTF8);
796 
797  // Save the original info.
798  mServer->registerChannelKey(sessionKey, origAddr.C());
799 
800  // Generate the address of the logger server.
801  QString addr = QString("%1:14666").arg(
802  mClientSocket->localAddress().toString() );
803 
804  // Generate the start game packet.
805  libcomp::Packet reply;
806  reply.WriteBlank(8);
807  reply.WriteU16Big(0x23);
808  reply.WriteU16Little(0x23);
809  reply.WriteU16Little(0x08);
810  reply.WriteU32Little(sessionKey);
811  reply.WriteString16Little(libcomp::Convert::ENCODING_UTF8,
812  addr.toUtf8().constData(), true);
813  reply.WriteU8(3);
814 
815  // Adjust the packet sizes.
816  reply.Seek(8);
817  reply.WriteU16Big((uint16_t)(reply.Size() - 10));
818  reply.WriteU16Little((uint16_t)(reply.Size() - 10));
819 
820  // Add a log message that indicates redirecting the client to the logger.
821  logMessage(tr("Sending client %1 to the logger...").arg(mClientID));
822 
823  // Encrypt the packet.
824  encryptCustomPacket(&mClientCryptData.key, reply);
825 
826  // Send the packet to the client.
827  mClientSocket->write((char*)reply.Data(), reply.Size());
828  mClientSocket->flush();
829 }
