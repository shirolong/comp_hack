278  QRegExp headerMatcher("([^\\:]+)\\:(.+)");
279

282  {
283  mClientSocket->disconnect();

285  return;
286  }

291  // If the content length isn't valid, close the connection.
292  if(contentSize < 1)
293  {
294  mClientSocket->disconnect();
295 
296  return;
297  }
298  }
299 
300  // If we don't have it all yet, keep waiting.
301  if(mBuffer.size() < contentSize)
302  return;
303 
304  // Where to copy the headers to.
305  QStringList headers;
306 
307  // Retrieve each hader and add it to the list.
308  foreach(QString key, mHeaderKeys)
309  headers.append( mHeaders.value(key) );
310 
311  // Send the HTTP headers to the target server.
312  mServerSocket->write( tr("%1\r\n%2\r\n\r\n").arg(mReqLine).arg(
313  headers.join("\r\n")).toLatin1() );

316  if(contentSize)
