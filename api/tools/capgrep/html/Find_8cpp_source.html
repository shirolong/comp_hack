127 void Find::findTerm()
128 {
129  // Get the search term entered.
130  QString term = ui.findEdit->text();
131 
132  // Make sure there is a search term.
133  if(term.isEmpty())
134  {
135  // Tell the user how stupid they are.
136  QMessageBox::critical(this, tr("Find Error"),
137  tr("You must enter a search term."));
138 
139  return;
140  }
141 
142  // Perform a different search based on the search type.
143  if(ui.binaryButton->isChecked())
144  {
145  // Remove spaces between the hex digits.
146  term.replace(" ", QString());
147 
148  // Make sure the term isn't empty now.
149  if(term.isEmpty())
150  {
151  // Tell the user how stupid they are.
152  QMessageBox::critical(this, tr("Find Error"),
153  tr("You must enter a search term."));
154 
155  return;
156  }
157 
158  // Since we are searching for complete bytes and each byte is two hex
159  // digits, the search term should have an even length. The search term
160  // should only contain valid hex values. We assume you are not going to
161  // write the hex prefix before any values (since we are not assuming
162  // any endianness and it's silly to add it for every byte).
163  if(term.length() % 2 != 0 || !QRegExp("[a-fA-F0-9]+").exactMatch(term))
164  {
165  QMessageBox::critical(this, tr("Find Error"), tr("A binary search "
166  "term must consist solely of a series of hex digit pairs."));
167 
168  return;
169  }
170 
171  QStringList labelText;
172  QByteArray binaryTerm;
173  uint8_t byte;
174 
175  // Convert each byte in the string into an integer. We can assume it
176  // will convert right since we checked for valid hex values above.
177  while(!term.isEmpty())
178  {
179  // Convert the byte.
180  byte = (uint8_t)term.left(2).toUInt(0, 16);
181 
182  // Remove the byte from the search term.
183  term = term.mid(2);
184 
185  // Add the byte to the binary seqeunce.
186  binaryTerm.append((char*)&byte, 1);
187 
188  // Add the byte to the search term label.
189  labelText.append(tr("%1").arg(
190  (uint32_t)byte & 0xFF, 2, 16, QLatin1Char('0')).toUpper());
191  }
192 
193  // Finish and set the search term label.
194  ui.searchLabel->setText(tr("Binary: %1").arg(labelText.join(" ")));
195 
196  // Find the binary sequence.
197  mFilter->findBinary(binaryTerm);
198  }
199  else if(ui.textButton->isChecked())
200  {
201  // Find the string using the desired encoding.
202  switch(ui.encoding->currentIndex())
203  {
204  case 0:
205  ui.searchLabel->setText(tr("Text (CP1252): %1").arg(term));
206 
207  mFilter->findText("CP1252", term);
208  break;
209  case 1:
210  ui.searchLabel->setText(tr("Text (CP932): %1").arg(term));
211 
212  mFilter->findText("CP932", term);
213  break;
214  case 2:
215  default:
216  ui.searchLabel->setText(tr("Text (UTF-8): %1").arg(term));
217 
218  mFilter->findText("UTF-8", term);
219  break;
220  }
221  }
