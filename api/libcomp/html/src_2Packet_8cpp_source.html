123 void Packet::GrowPacket(uint32_t sz)
124 {
125  // Allocate the packet data (if needed).
126  Allocate();
127 
128  // Make sure the packet is growing.
129  if(0 == sz)
130  {
131  PACKET_EXCEPTION("Attempted to grow the packet by 0 bytes", this);
132  }
133 
134  // Calculate how big the packet will be if it can grow.
135  uint32_t newSize = mPosition + sz;
136 
137  /* If the new size of the packet is less than the current size, keep the
138  * packet at it's current size. If the new packet size is too big, throw a
139  * packet exception. If the size is within the max packet size, set the new
140  * size of the packet.
141  */
142  if(newSize < mSize)
143  {
144  // Keep the current size, do nothing.
145  }
146  else if(MAX_PACKET_SIZE < newSize || MAX_PACKET_SIZE < (mPosition + sz))
147  {
148  // The new size is too big, throw an exception.
149  PACKET_EXCEPTION(String("Attempted to grow packet by %1 bytes; "
150  "however, doing so would exceed MAX_PACKET_SIZE").Arg(sz), this);
151  }
152  else
153  {
154  // The new packet size is valid, set it.
155  mSize = newSize;
156  }
157 }
