207 bool ChannelConnection::DecompressPacket(libcomp::Packet& packet,
208  uint32_t& paddedSize, uint32_t& realSize, uint32_t& dataStart)
209 {
210  // Make sure we are at the right spot (right after the sizes).
211  packet.Seek(2 * sizeof(uint32_t));
212 
213  // All packets that support compression have this.
214  if(0x677A6970 != packet.ReadU32Big()) // "gzip"
215  {
216  SocketError();
217 
218  return false;
219  }
220 
221  // Read the sizes.
222  int32_t uncompressedSize = packet.ReadS32Little();
223  int32_t compressedSize = packet.ReadS32Little();
224 
225  // Sanity check the sizes.
226  if(0 > uncompressedSize || 0 > compressedSize)
227  {
228  SocketError();
229 
230  return false;
231  }
232 
233  // Check that the compression is as expected.
234  if(0x6C763600 != packet.ReadU32Big()) // "lv6\0"
235  {
236  SocketError();
237 
238  return false;
239  }
240 
241  // Calculate how much data is padding
242  uint32_t padding = paddedSize - realSize;
243 
244  // Make sure the packet is the right size.
245  if(packet.Left() != (static_cast<uint32_t>(compressedSize) + padding))
246  {
247  SocketError();
248 
249  return false;
250  }
251 
252  // Only decompress if the sizes are not the same.
253  if(compressedSize != uncompressedSize)
254  {
255  // Attempt to decompress.
256  int32_t sz = packet.Decompress(compressedSize);
257 
258  // Check the uncompressed size matches the recorded size.
259  if(sz != uncompressedSize)
260  {
261  SocketError();
262 
263  return false;
264  }
265 
266  // The is no padding anymore.
267  paddedSize = realSize = static_cast<uint32_t>(sz);
268  }
269 
270  // Skip over: gzip, lv0, uncompressedSize, compressedSize.
271  dataStart += static_cast<uint32_t>(sizeof(uint32_t) * 4);
272 
273  return true;
