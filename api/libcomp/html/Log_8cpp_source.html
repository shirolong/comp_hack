363 void Log::LogMessage(const std::chrono::time_point<
364  std::chrono::system_clock>& now, LogComponent_t comp,
365  Log::Level_t level, const String& msg)
366 {
367  // Prepend these to messages.
368  static const String gLogMessages[Log::LOG_LEVEL_COUNT] = {
369  "DEBUG: %1%2",
370  "%1%2",
371  "WARNING: %1%2",
372  "ERROR: %1%2",
373  "CRITICAL: %1%2",
374  };
375 
376  // Log a critical error message. If the configuration option is true, log
377  // the message to the log file. Regardless, pass the message to all the
378  // log hooks for processing. Critical messages have the text "CRITICAL: "
379  // appended to them.
380  if(!ShouldLog(comp, level))
381  {
382  return;
383  }
384 
385  String compStr = LogComponentToString(comp);
386 
387  if(!compStr.IsEmpty())
388  {
389  compStr += ": ";
390  }
391 
392  String final = String(gLogMessages[level]).Arg(compStr).Arg(msg);
393 
394  if(nullptr != mLogFile)
395  {
396  auto duration = std::chrono::duration_cast< std::chrono::duration<
397  int64_t, std::ratio<86400>> >( now.time_since_epoch() ).count();
398 
399  if(mLogRotationEnabled && mLogRotationDays <= (duration - mLastLog))
400  {
401  RotateLogs();
402  }
403 
404  mLastLog = duration;
405 
406  if(mLogFileTimestampEnabled)
407  {
408  auto currentTime = std::chrono::system_clock::to_time_t(now);
409 
410  std::stringstream ss;
411  ss << std::put_time(std::localtime(&currentTime), "%Y/%m/%d %T");
412 
413  String formattedTime = String("[%1] ").Arg(ss.str());
414 
415  mLogFile->write(formattedTime.C(),
416  (std::streamsize)(formattedTime.Size() * sizeof(char)));
417  }
418 
419  mLogFile->write(final.C(),
420  (std::streamsize)(final.Size() * sizeof(char)));
421  mLogFile->flush();
422  }
423 
424  // Call all hooks.
425  for(auto i : mHooks)
426  {
427  (*i.first)(comp, level, final, i.second);
428  }
429 
430  // Call all lambda hooks.
431  for(auto func : mLambdaHooks)
432  {
433  func(comp, level, final);
434  }

594 void Log::RotateLogs()
595 {
596  // Do not rotate if the main log does not exist.
597  if(!mLogPath.IsEmpty() && !FileExists(mLogPath))
598  {
599  return;
600  }
601 
602  // Delete the last log that we will rotate out.
603  FileDelete(libcomp::String("%1.%2").Arg(mLogPath).Arg(
604  mLogRotationCount));
605  FileDelete(libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(
606  mLogRotationCount));
607 
608  // Now we need to rotate them out moving the files as we go.
609  for(int i = mLogRotationCount - 1; i > 0; --i)
610  {
611  FileMove(libcomp::String("%1.%2").Arg(mLogPath).Arg(i),
612  libcomp::String("%1.%2").Arg(mLogPath).Arg(i + 1));
613  FileMove(libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(i),
614  libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(i + 1));
615  }
616 
617  // Now close the main log and move it.
618  if(mLogFile)
619  {
620  mLogFile->close();
621  delete mLogFile;
622  }
623 
624  FileMove(mLogPath, mLogPath + ".1");
625 
626  // Open the log again.
627  mLogFile = new std::ofstream();
628  mLogFile->open(mLogPath.C(), std::ofstream::out | std::ofstream::trunc);
629  mLogFile->flush();
630 
631  // If this failed, close it.
632  if(!mLogFile->good())
633  {
634  delete mLogFile;
635  mLogFile = nullptr;
636  mLogPath.Clear();
637  }
638 
639  // If compression is enabled compress the log we just moved.
640  if(mLogCompression)
641  {
642  std::thread th([](libcomp::String compressPath)
643  {
644  bool error = false;
645  char buffer[4096];
646  libcomp::String compressedPath = compressPath + ".gz";
647 
648  FILE *in = fopen(compressPath.C(), "rb");
649  gzFile gf = gzopen(compressedPath.C(), "wb");
650 
651  if(in && gf)
652  {
653  while(!feof(in))
654  {
655  auto sz = fread(buffer, 1, sizeof(buffer), in);
656 
657  if(0 >= sz)
658  {
659  break;
660  }
661 
662  if((int)sz != gzwrite(gf, buffer, (unsigned)sz))
663  {
664  error = true;
665  break;
666  }
667  }
668  }
669 
670  gzclose(gf);
671  fclose(in);
672 
673  if(error)
674  {
675  FileDelete(compressedPath);
676  }
677  else
678  {
679  FileDelete(compressPath);
680  }
681  }, mLogPath + ".1");
682 
683  // Detach the thread and let it work on it's own.
684  th.detach();
685  }
686 }
