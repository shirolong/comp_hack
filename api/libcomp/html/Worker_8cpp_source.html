107 {
108  // Check for a shutdown message.
109  libcomp::Message::Shutdown *pShutdown = dynamic_cast<
110  libcomp::Message::Shutdown*>(pMessage);
111 
112  // Check for an execute message.
113  libcomp::Message::Execute *pExecute = dynamic_cast<
114  libcomp::Message::Execute*>(pMessage);
115 
116  // Do not handle any more messages if a shutdown was sent.
117  if(nullptr != pShutdown || !mRunning)
118  {
119  mRunning = false;
120  }
121  else if(nullptr != pExecute)
122  {
123  // Run the code now.
124  pExecute->Run();
125  }
126  else
127  {

129 
130  // Attempt to find a manager to process this message.
131  auto range = mManagers.equal_range(pMessage->GetType());
132 
133  // List of managers to process for.
134  std::list<std::shared_ptr<Manager>> managers;
135 
136  // Get the list of managers to process.
137  for(auto it = range.first; it != range.second; ++it)
138  {
139  auto manager = it->second;
140 
141  if(!manager)

143  LogGeneralErrorMsg("Manager is null!\n");
144  }
145  else
146  {
147  managers.push_back(manager);
148  }
149  }
150 
151  // Process the message with the list of managers.
152  for(auto manager : managers)
153  {
154  if(manager->ProcessMessage(pMessage))

158  }
