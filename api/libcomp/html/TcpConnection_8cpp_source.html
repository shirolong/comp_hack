180 bool TcpConnection::RequestPacket(size_t size)
181 {
182  bool result = false;
183 
184  // Make sure the buffer is there.
185  mReceivedPacket.Allocate();
186 
187  if(0 < mReceivedPacket.Size())
188  {
189  LogConnectionDebug([&]()
190  {
191  return String("TcpConnection::RequestPacket() called when there is "
192  "still %1 bytes in the buffer.\n").Arg(mReceivedPacket.Size());
193  });
194  }
195 
196  // Get direct access to the buffer.
197  char *pDestination = mReceivedPacket.Data();
198 
199  if(0 != size && MAX_PACKET_SIZE >= (mReceivedPacket.Size() + size) &&
200  nullptr != pDestination)
201  {
202  // Calculate where to write the data.
203  pDestination += mReceivedPacket.Size();
204 
205  // Get a shared pointer to the connection so it outlives the callback.
206  auto self = shared_from_this();
207 
208  // Request packet data from the socket.
209  mSocket.async_receive(asio::buffer(pDestination, size), 0,
210  [self](asio::error_code errorCode, std::size_t length)
211  {
212  if(errorCode)
213  {
214  LogConnectionDebug([&]()
215  {
216  return String("ASIO Error: %1\n")
217  .Arg(errorCode.message());
218  });
219 
220  self->SocketError();
221  }
222  else
223  {
224  // Adjust the size of the packet.
225  (void)self->mReceivedPacket.Direct(
226  self->mReceivedPacket.Size() +
227  static_cast<uint32_t>(length));
228  self->mReceivedPacket.Rewind();
229 
230  // It's up to this callback to remove the data from the
231  // packet either by calling std::move() or packet.Clear().
232  self->PacketReceived(self->mReceivedPacket);
233 
234  if(0 < self->mReceivedPacket.Size())
235  {
236  LogConnectionDebug([&]()
237  {
238  return String("TcpConnection::PacketReceived() was "
239  "called and it left %1 bytes in the buffer.\n"
240  ).Arg(self->mReceivedPacket.Size());
241  });
242  }
243  }
244  });
245 
246  // Success.
247  result = true;
248  }
249 
250  return result;
251 }

333 void TcpConnection::FlushOutgoingInside(bool closeConnection)
334 {
335  // Don't send anything if we are not connected.
336  if(STATUS_NOT_CONNECTED == mStatus)
337  {
338  return;
339  }
340 
341  // Get a shared pointer to the connection so it outlives the callback.
342  auto self = shared_from_this();
343 
344  mSocket.async_send(asio::buffer(mOutgoing.ConstData() +
345  mOutgoing.Tell(), mOutgoing.Left()), 0, [closeConnection,
346  self](asio::error_code errorCode, std::size_t length)
347  {
348  bool sendSame = false;
349  bool sendAnother = false;
350  bool packetOk = false;
351 
352  ReadOnlyPacket readOnlyPacket;
353 
354  // Ignore errors and everything else, just close the connection.
355  if(closeConnection && self->mOutgoing.Left() == length)
356  {
357  LogConnectionDebugMsg("Closing connection after sending packet.\n");
358 
359  std::lock_guard<std::mutex> outgoingGuard(
360  self->mOutgoingMutex);
361 
362  self->mSendingPacket = false;
363 
364  self->SocketError();
365  return;
366  }
367 
368  if(errorCode)
369  {
370  std::lock_guard<std::mutex> outgoingGuard(
371  self->mOutgoingMutex);
372 
373  self->mSendingPacket = false;
374 
375  self->SocketError();
376  }
377  else
378  {
379  std::lock_guard<std::mutex> outgoingGuard(self->mOutgoingMutex);
380 
381  uint32_t outgoingSize = self->mOutgoing.Size();
382 
383  if(0 == outgoingSize || length > self->mOutgoing.Left())
384  {
385  self->SocketError();
386  }
387  else
388  {
389  self->mOutgoing.Skip((uint32_t)length);
390 
391  if(0 != self->mOutgoing.Left())
392  {
393  sendSame = true;
394  }
395  else
396  {
397  self->mOutgoing.Rewind();
398 
399  readOnlyPacket = self->mOutgoing;
400  sendAnother = !self->mOutgoingPackets.empty();
401  packetOk = true;
402  }
403  }
404 
405  self->mSendingPacket = sendSame;
406  }
407 
408  if(sendSame)
409  {
410  self->FlushOutgoingInside(closeConnection);
411  }
412  else if(packetOk)
413  {
414  self->PacketSent(readOnlyPacket);
415 
416  if(sendAnother)
417  {
418  self->FlushOutgoing();
419  }
420  }
421  });
