668 bool BaseServer::LoadDataFromFile(const libcomp::String& filePath,

671 {

674  {
675  return false;
676  }

681  });
682 
683  const tinyxml2::XMLElement *objXml = doc.RootElement()->FirstChildElement("object");
684 
685  while(nullptr != objXml)
686  {
687  bool typeExists = true;
688

691  {
692  continue;
693  }
694

696  if(!typeExists)

723  }

726  if(!record->Load(doc, *objXml))

733 
734  return false;
735  }

739  // Check that there is a password and salt it

741

752  account->SetSalt(salt);
753  account->SetPassword(libcomp::Crypto::HashPassword(
754  account->GetPassword(), salt));
755  }
756 
757  if(registerRecords && !record->Register(record, uuid))
758  {
759  return false;
760  }
761  else if(!registerRecords)
762  {
763  record->Unregister();
764  }
765

767

769  }
