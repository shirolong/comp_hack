181 DatabaseBind* DBExplicitUpdate::Verify(const libcomp::String& column,
182  const std::set<libobjgen::MetaVariable::MetaVariableType_t>& validTypes)
183 {
184  auto key = column.C();
185  auto var = mMetadata ? mMetadata->GetVariable(key) : nullptr;
186  auto existing = mChanges.find(key);
187  auto stored = mStoredValues.find(key);
188  auto expected = mExpectedValues.find(key);
189 
190  return ((var && validTypes.find(var->GetMetaType()) != validTypes.end())
191  && existing == mChanges.end()
192  && expected == mExpectedValues.end()
193  && stored != mStoredValues.end()) ? stored->second : 0;
194 }
