607 void EncryptedConnection::ParsePacket(libcomp::Packet& packet,
608  uint32_t paddedSize, uint32_t realSize)
609 {
610  // Decrypt the packet
611  mEncryptionKey.DecryptPacket(packet);
612 
613  // Save the packet to the capture.
614  if(nullptr != mCaptureFile)
615  {
616  std::time_t now = std::time(nullptr);
617 
618  uint8_t source = HACK_SOURCE_CLIENT;
619  uint64_t stamp = static_cast<uint64_t>(now);
620  uint64_t microtime = static_cast<uint64_t>(
621  std::chrono::time_point_cast<std::chrono::microseconds>(
622  std::chrono::steady_clock::now()
623  ).time_since_epoch().count());
624  uint32_t size = packet.Size();
625 
626  mCaptureFile->write(reinterpret_cast<char*>(&source),
627  sizeof(source));
628  mCaptureFile->write(reinterpret_cast<char*>(&stamp),
629  sizeof(stamp));
630  mCaptureFile->write(reinterpret_cast<char*>(&microtime),
631  sizeof(microtime));
632  mCaptureFile->write(reinterpret_cast<char*>(&size),
633  sizeof(size));
634  mCaptureFile->write(packet.ConstData(),
635  size);
636  mCaptureFile->flush();
637 
638  if(!mCaptureFile->good())
639  {
640  LogConnectionCriticalMsg("Failed to write capture file.\n");
641 
642  delete mCaptureFile;
643  mCaptureFile = nullptr;
644  }
645  }
646 
647  // This is where to find the data.
648  uint32_t dataStart = 2 * sizeof(uint32_t);
649 
650  // Decompress the packet.
651  if(!DecompressPacket(packet, paddedSize, realSize, dataStart))
652  {
653  return;
654  }
655 
656  // Move the packet into a read only copy.
657  ReadOnlyPacket copy(std::move(packet));
658 
659  // Make sure we are at the right spot (right after the sizes).
660  copy.Seek(dataStart);
661 
662  // Calculate how much data is padding.
663  uint32_t padding = paddedSize - realSize;
664 
665  // This will stop the command parsing.
666  bool errorFound = false;
667 
668  // Keep reading each command (sometimes called a packet) inside the
669  // decrypted packet from the network socket.
670  while(!errorFound && copy.Left() > padding)
671  {
672  // Make sure there is enough data
673  if(copy.Left() < 3 * sizeof(uint16_t))
674  {
675  SocketError("Corrupt packet (not enough data for command header).");
676 
677  errorFound = true;
678  }
679  else
680  {
681  // Skip over the big endian size (we think).
682  copy.Skip(2);
683 
684  // Remember where this command started so we may advance over it
685  // after it has been parsed.
686  uint32_t commandStart = copy.Tell();
687  uint16_t commandSize = copy.ReadU16Little();
688  uint16_t commandCode = copy.ReadU16Little();
689 
690  // With no data, the command size is 4 bytes (code + a size).
691  if(commandSize < 2 * sizeof(uint16_t))
692  {
693  SocketError("Corrupt packet (not enough data for command).");
694 
695  errorFound = true;
696  }
697 
698  // Check there is enough packet left for the command data.
699  if(!errorFound && copy.Left() < (uint32_t)(commandSize -
700  2 * sizeof(uint16_t)))
701  {
702  SocketError("Corrupt packet (not enough data for "
703  "command data).");
704 
705  errorFound = true;
706  }
707 
708  // Check for the message queue.
709  auto messageQueue = mMessageQueue.lock();
710 
711  if(!errorFound && nullptr == messageQueue)
712  {
713  SocketError("No message queue for packet.");
714 
715  errorFound = true;
716  }
717 
718  // Promote to a shared pointer.
719  auto self = shared_from_this();
720 
721  if(!errorFound && this != self.get())
722  {
723  SocketError("Failed to obtain a shared pointer.");
724 
725  errorFound = true;
726  }
727 
728  if(!errorFound)
729  {
730  // This is a shallow copy of the command data.
731  ReadOnlyPacket command(copy, commandStart +
732  2 * static_cast<uint32_t>(sizeof(uint16_t)),
733  commandSize - 2 * static_cast<uint32_t>(
734  sizeof(uint16_t)));
735 
736  // Notify the task about the new packet.
737  messageQueue->Enqueue(new libcomp::Message::Packet(self,
738  commandCode, command));
739  }
740 
741  // Move to the next command.
742  if(!errorFound)
743  {
744  copy.Seek(commandStart + commandSize);
745  }
746  }
747  } // while(!errorFound && packet.Left() > padding)
748 
749  if(!errorFound)
750  {
751  // Skip the padding
752  copy.Skip(padding);
753  }
754 
755  if(!errorFound && copy.Left() != 0)
756  {
757  SocketError("Corrupt packet has extra data.");
758  }
759 }
