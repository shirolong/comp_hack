802 String ReadOnlyPacket::Dump() const
803 {
804  // Do not generate a hex dump if there is not data.
805  if(0 == mSize)
806  {
807  return String();
808  }
809 
810  uint32_t line = 0;
811 
812  std::list<String> final;
813 
814  // Loop for each line of the hex dump.
815  while(line < mSize)
816  {
817  libcomp::String buffer;
818 
819  // Print line number (add it to the working buffer). If the current
820  // position is on the first byte of this line, print the left side of
821  // the position marker.
822  buffer += libcomp::String(mPosition == line ? "%1 {" : "%1 ").Arg(
823  line, 4, 16, '0');
824 
825  // Loop and print 8 bytes of the ReadOnlypacket. If the current byte is past
826  // the end of the ReadOnlypacket, blank spaces will be written. If the current
827  // position is the next byte, print the left side of the position
828  // marker. If the current byte is the current position, print the byte
829  // followed by the right size of the position marker. Also, place a
830  // single space between each byte in the dump.
831  for(uint32_t i = line; i < (line + 8); ++i)
832  {
833  if(i >= mSize)
834  {
835  buffer += mPosition == i ? " }" : " ";
836  }
837  else if(mPosition == i)
838  {
839  buffer += libcomp::String("%1}").Arg(mData[i], 2, 16, '0');
840  }
841  else if(mPosition == (i + 1) && (i + 1) != (line + 8))
842  {
843  buffer += libcomp::String("%1{").Arg(mData[i], 2, 16, '0');
844  }
845  else
846  {
847  buffer += libcomp::String("%1 ").Arg(mData[i], 2, 16, '0');
848  }
849  }
850 
851  // Place a gap between the two sets of 8 bytes. Also place the left
852  // side of the current position marker if the current position is the
853  // first byte in the next set of 8 bytes.
854  buffer += mPosition == (line + 8) ? "{" : " ";
855 
856  // Loop and print 8 bytes of the ReadOnlypacket. If the current byte is past
857  // the end of the ReadOnlypacket, blank spaces will be written. If the current
858  // position is the next byte, print the left side of the position
859  // marker. If the current byte is the current position, print the byte
860  // followed by the right size of the position marker. Also, place a
861  // single space between each byte in the dump.
862  for(uint32_t i = (line + 8); i < (line + 16); ++i)
863  {
864  if(i >= mSize)
865  {
866  buffer += mPosition == i ? " }" : " ";
867  }
868  else if(mPosition == i)
869  {
870  buffer += libcomp::String("%1}").Arg(mData[i], 2, 16, '0');
871  }
872  else if(mPosition == (i + 1) && (i + 1) != (line + 16))
873  {
874  buffer += libcomp::String("%1{").Arg(mData[i], 2, 16, '0');
875  }
876  else
877  {
878  buffer += libcomp::String("%1 ").Arg(mData[i], 2, 16, '0');
879  }
880  }
881 
882  // Print a space between the hex dump and the ASCII representation.
883  buffer += " ";
884 
885  // Print the ASCII character the the byte represents. Only printable
886  // ASCII is included; all other characters are printed as a dot.
887  for(uint32_t i = line; i < (line + 8) && i < mSize; ++i)
888  {
889  uint8_t val = mData[i];
890 
891  buffer += libcomp::String(1, (val >= 0x20 && val < 0x7f) ?
892  static_cast<char>(val) : '.');
893  }
894 
895  // Print a space between the two sets of 8 ASCII characters.
896  buffer += " ";
897 
898  // Print the ASCII character the the byte represents. Only printable
899  // ASCII is included; all other characters are printed as a dot.
900  for(uint32_t i = (line + 8); i < (line + 16) && i < mSize; ++i)
901  {
902  uint8_t val = mData[i];
903 
904  buffer += libcomp::String(1, (val >= 0x20 && val < 0x7f) ?
905  static_cast<char>(val) : '.');
906  }
907 
908  // Advance the line by 16 bytes
909  line += 16;
910 
911  // Append the generated line to the output string.
912  final.push_back(buffer);
913  }
914 
915  // If the cursor is at the end and there is a multiple of 16 bytes in the
916  // packet, make another line for the cursor position.
917  if(mPosition == mSize && 0 == (mSize % 16))
918  {
919  final.push_back(libcomp::String("%1 { }").Arg(line, 4, 16, '0'));
920  }
921 
922  // Join all the lines of the hex dump together and return the final string.
923  return String::Join(final, "\n");
924 }
