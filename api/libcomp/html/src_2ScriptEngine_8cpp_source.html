273 bool ScriptEngine::Eval(const String& source, const String& sourceName)
274 {
275  bool result = false;
276 
277  SQInteger top = sq_gettop(mVM);
278 
279  if(SQ_SUCCEEDED(sq_compilebuffer(mVM, source.C(),
280  (SQInteger)source.Size(), sourceName.C(), 1)))
281  {
282  sq_pushroottable(mVM);
283 
284  if(SQ_SUCCEEDED(sq_call(mVM, ONE_PARAM,
285  NO_RETURN_VALUE, RAISE_ERROR)))
286  {
287  result = true;
288  }
289  }
290 
291  sq_settop(mVM, top);

293  return result;
