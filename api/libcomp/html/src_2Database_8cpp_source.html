132 bool Database::QueueChangeSet(const std::shared_ptr<
133  DatabaseChangeSet>& changes)
134 {
135  auto uuid = changes->GetTransactionUUID();
136  std::string key = uuid.ToString();
137 
138  auto opChanges = std::dynamic_pointer_cast<
139  DBOperationalChangeSet>(changes);
140  if(opChanges)
141  {
142  // Operational change queuing is not supported
143  return false;
144  }
145  else
146  {
147  auto standardChanges = std::dynamic_pointer_cast<
148  DBStandardChangeSet>(changes);
149 
150  if(standardChanges)
151  {
152  std::lock_guard<std::mutex> lock(mTransactionLock);
153  auto queueEntry = mTransactionQueue[key];
154  if(queueEntry == nullptr)
155  {
156  queueEntry = std::make_shared<DBStandardChangeSet>(uuid);
157  }
158 
159  for(auto obj : standardChanges->GetInserts())
160  {
161  queueEntry->Insert(obj);
162  }
163 
164  for(auto obj : standardChanges->GetUpdates())
165  {
166  queueEntry->Update(obj);
167  }
168 
169  for(auto obj : standardChanges->GetDeletes())
170  {
171  queueEntry->Delete(obj);
172  }
173 
174  mTransactionQueue[key] = queueEntry;
175 
176  return true;
177  }
178  }
179 
180  return false;
181 }

183 std::list<libobjgen::UUID> Database::ProcessTransactionQueue()
184 {
185  std::list<libobjgen::UUID> failures;
186 
187  std::unordered_map<std::string,
188  std::shared_ptr<DBStandardChangeSet>> queue;
189  {
190  std::lock_guard<std::mutex> lock(mTransactionLock);
191 
192  if(mTransactionQueue.size() == 0)
193  {
194  return failures;
195  }
196 
197  queue = mTransactionQueue;
198  mTransactionQueue.clear();
199  }
200 
201  // Process the general queue transaction first
202  auto nullKey = NULLUUID.ToString();
203  if(queue.find(nullKey) != queue.end())
204  {
205  if(!ProcessChangeSet(queue[nullKey]))
206  {
207  failures.push_back(nullKey);
208  }
209  queue.erase(nullKey);
210  }
211 
212  // Process the remaining transactions
213  for(auto kv : queue)
214  {
215  if(!ProcessChangeSet(kv.second))
216  {
217  failures.push_back(kv.second->GetTransactionUUID());
218  }
219  }
220 
221  return failures;
222 }
