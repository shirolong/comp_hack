114 bool PersistentObject::Register(const std::shared_ptr<PersistentObject>& self,
115  const libobjgen::UUID& pUuid)
116 {
117  if(!self->IsDeleted())
118  {
119  bool registered = false;
120 
121  libobjgen::UUID& uuid = self->mUUID;
122 
123  std::lock_guard<std::mutex> lock(mCacheLock);
124  if(!pUuid.IsNull() && !uuid.IsNull())
125  {
126  // Unregister old UUID, keep if making a copy
127  auto it = sCached.find(uuid.ToString());
128  if(it != sCached.end() && it->second.lock() == self)
129  {
130  sCached.erase(it);
131  }
132  }
133 
134  if(!pUuid.IsNull())
135  {
136  uuid = pUuid;
137  }
138  else if(uuid.IsNull())
139  {
140  uuid = libobjgen::UUID::Random();
141  registered = true;
142  }
143 
144  if(!registered && sCached.find(uuid.ToString()) == sCached.end())
145  {
146  registered = true;
147  }
148 
149  std::string uuidString = uuid.ToString();
150 
151  if(registered)
152  {
153  self->mSelf = self;
154  sCached[uuidString] = self;
155 
156  return true;
157  }
158  else
159  {
160  LogGeneralError([&]()
161  {
162  return String("Duplicate object detected: %1\n")
163  .Arg(uuidString);
164  });
165  }
166  }
167 
168  return false;
169 }
