411  bool LoadObjectsFromFile(gsl::not_null<DataStore*> pDataStore,
412  const libcomp::String& filePath,
413  DefinitionManager* definitionManager = nullptr)
414  {
415  tinyxml2::XMLDocument objsDoc;
416 
417  std::vector<char> data = pDataStore->ReadFile(filePath);
418 
419  if(data.empty())
420  {
421  LogServerDataManagerWarning([&]()
422  {
423  return String("File does not exist or is empty: %1\n")
424  .Arg(filePath);
425  });
426 
427  return true;
428  }
429 
430  if(tinyxml2::XML_SUCCESS !=
431  objsDoc.Parse(&data[0], data.size()))
432  {
433  return false;
434  }
435 
436  const tinyxml2::XMLElement *rootNode = objsDoc.RootElement();
437  const tinyxml2::XMLElement *objNode = rootNode->FirstChildElement("object");
438 
439  while(nullptr != objNode)
440  {
441  if(!LoadObject<T>(objsDoc, objNode, definitionManager))
442  {
443  LogServerDataManagerError([&]()
444  {
445  return String("Failed to load XML file: %1\n")
446  .Arg(filePath);
447  });
448 
449  return false;
450  }
451 
452  objNode = objNode->NextSiblingElement("object");
453  }
454 
455  LogServerDataManagerInfo([&]()
456  {
457  return String("Loaded XML file: %1\n").Arg(filePath);
458  });
459 
460  return true;
461  }
