278 void rbtree_delete(rbtree t, rbtree_key key, compare_func compare) {
279  node child;
280  node n = lookup_node(t, key, compare);
281  if (n == NULL) return; /* Key not found, do nothing */
282  if (n->left != NULL && n->right != NULL) {
283  /* Copy key/value from predecessor and then delete it instead */
284  node pred = maximum_node(n->left);
285  n->key = pred->key;
286  n->value = pred->value;
287  n = pred;
288  }
289 
290  assert(n->left == NULL || n->right == NULL);
291  child = n->right == NULL ? n->left : n->right;
292  if (node_color(n) == BLACK) {
293  n->color = node_color(child);
294  delete_case1(t, n);
295  }
296  replace_node(t, n, child);
297  if (n->parent == NULL && child != NULL) // root should be black
298  child->color = BLACK;
299  free(n);
300 
301  verify_properties(t);
302 }
